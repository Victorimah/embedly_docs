### Request Body

<ParamField body="id" type="string" required>
	The unique identifier for the organization.  
	**Example:** '3fa85f64-5717-4562-b3fc-2c963f66afa6'
</ParamField>

<ParamField body="name" type="string" required>
	The name of the organization.  
	**Example:** 'Eco'
</ParamField>

<ParamField body="key" type="string" required>
	A key associated with the organization.  
	**Example:** '66666666666666666666'
</ParamField>

<ParamField body="dateCreated" type="string" required>
	The date and time the organization was created (ISO 8601 format).  
	**Example:** '2025-04-06T17:02:46.852Z'
</ParamField>

<ParamField body="walletTypeId" type="string" required>
	The unique identifier for the wallet type associated with the organization.  
	**Example:** '3fa85f64-5717-4562-b3fc-2c963f66afa6'
</ParamField>

<ParamField body="virtualAccountTypeId" type="string" required>
	The unique identifier for the virtual account type associated with the organization.  
	**Example:** '3fa85f64-5717-4562-b3fc-2c963f66afa6'
</ParamField>

<ParamField body="prefixes" type="array[string]" required>
	An array of prefixes associated with the organization.  
	**Example:** ['string']
</ParamField>

<ParamField body="walletType" type="string" required>
	The type of wallet for the organization.  
	**Example:** 'Open'
</ParamField>

<ParamField body="virtualAccountType" type="string" required>
	The type of virtual account for the organization.  
	**Example:** 'string'
</ParamField>
